4 linhas dentro do while
def reverseList(self, head):
    prev = None       # lista invertida (começa vazia)
    curr = head      # nó actual
 
    while curr is not None:
        next_node = curr.next   # ① guardar próximo
        curr.next = prev      # ② inverter ponteiro
        prev = curr          # ③ avançar prev
        curr = next_node     # ④ avançar curr
 
    return prev           # novo head
